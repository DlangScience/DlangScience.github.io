    Function toArgv

    Convert D's string[] args to a correct C char** argv.

    Prototype

    char** toArgv(
      string[] args
    );

    Example

    auto args = ["hello", "world"];
auto argv = args.toArgv;

foreach(i, arg; args)
{
    assert(argv[i][0 .. args[i].length] == args[i]);
    assert(argv[i][args[i].length] == 0);
}
assert(argv[args.length] == null);
